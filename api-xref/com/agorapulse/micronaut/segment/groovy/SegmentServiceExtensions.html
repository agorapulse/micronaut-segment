18  package com.agorapulse.micronaut.segment.groovy;
19  
20  import com.agorapulse.micronaut.segment.SegmentService;
21  import groovy.lang.Closure;
22  import groovy.lang.DelegatesTo;
23  import groovy.transform.stc.ClosureParams;
24  import groovy.transform.stc.FromString;
25  import space.jasan.support.groovy.closure.ConsumerWithDelegate;
26  
27  public class SegmentServiceExtensions {
28  
29      private SegmentServiceExtensions() { }
30  
31      /**
32       * Alias method lets you merge two user profiles, including their actions and traits.
33       *
34       * @param from    the user's id after they are logged in. It's the same id as
35       *                which you would recognize a signed-in user in your system.
36       * @param builder Consumer of the builder for the additional configuration
37       */
38      public static void alias(
39          SegmentService self,
40          String from,
41          String to,
42          @DelegatesTo(type = "com.agorapulse.micronaut.segment.builder.SimpleMessageBuilder", strategy = Closure.DELEGATE_FIRST)
43          @ClosureParams(value = FromString.class, options = "com.agorapulse.micronaut.segment.builder.SimpleMessageBuilder")
44              Closure<?> builder
45      ) {
46          self.alias(from, to, ConsumerWithDelegate.create(builder));
47      }
48  
49      /**
50       * Group method lets you associate a user with a group.
51       *
52       * @param userId  the user's id after they are logged in. It's the same id as
53       *                which you would recognize a signed-in user in your system.
54       * @param groupId The ID for this group in your database.
55       * @param builder Consumer of the builder for the additional configuration
56       */
57      public static void group(
58          SegmentService self,
59          String userId,
60          String groupId,
61          @DelegatesTo(type = "com.agorapulse.micronaut.segment.builder.MessageBuilderWithTraits", strategy = Closure.DELEGATE_FIRST)
62          @ClosureParams(value = FromString.class, options = "com.agorapulse.micronaut.segment.builder.MessageBuilderWithTraits")
63              Closure<?> builder
64      ) {
65          self.group(userId, groupId, ConsumerWithDelegate.create(builder));
66      }
67  
68  
69      /**
70       * identify lets you tie a user to their actions and record traits about them.
71       *
72       * @param userId  The ID for this user in your database.
73       * @param builder Consumer of the builder for the additional configuration
74       */
75      public static void identify(
76          SegmentService self,
77          String userId,
78          @DelegatesTo(type = "com.agorapulse.micronaut.segment.builder.MessageBuilderWithTraits", strategy = Closure.DELEGATE_FIRST)
79          @ClosureParams(value = FromString.class, options = "com.agorapulse.micronaut.segment.builder.MessageBuilderWithTraits")
80              Closure<?> builder
81      ) {
82          self.identify(userId, ConsumerWithDelegate.create(builder));
83      }
84  
85      /**
86       * Page method lets you record webpage visits from your web servers.
87       *
88       * @param userId  The ID for this user in your database.
89       * @param name    The webpage name you’re tracking. We recommend human-readable
90       *                names like Login or Register.
91       * @param builder Consumer of the builder for the additional configuration
92       */
93      public static void page(
94          SegmentService self,
95          String userId,
96          String name,
97          @DelegatesTo(type = "com.agorapulse.micronaut.segment.builder.MessageBuilderWithProperties", strategy = Closure.DELEGATE_FIRST)
98          @ClosureParams(value = FromString.class, options = "com.agorapulse.micronaut.segment.builder.MessageBuilderWithProperties")
99              Closure<?> builder
100     ) {
101         self.page(userId, name, ConsumerWithDelegate.create(builder));
102     }
103 
104     /**
105      * screen lets you record mobile screen views from your web servers.
106      *
107      * @param userId  The ID for this user in your database.
108      * @param name    The screen name you’re tracking. We recommend human-readable
109      *                names like Login or Register.
110      * @param builder Consumer of the builder for the additional configuration
111      */
112     public static void screen(
113         SegmentService self,
114         String userId,
115         String name,
116         @DelegatesTo(type = "com.agorapulse.micronaut.segment.builder.MessageBuilderWithProperties", strategy = Closure.DELEGATE_FIRST)
117         @ClosureParams(value = FromString.class, options = "com.agorapulse.micronaut.segment.builder.MessageBuilderWithProperties")
118             Closure<?> builder
119     ) {
120         self.screen(userId, name, ConsumerWithDelegate.create(builder));
121     }
122 
123     /**
124      * track lets you record the actions your users perform.
125      *
126      * @param userId  The ID for this user in your database.
127      * @param event   The name of the event you’re tracking. We recommend human-readable
128      *                names like Played Song or Updated Status.
129      * @param builder Consumer of the builder for the additional configuration
130      */
131     public static void track(
132         SegmentService self,
133         String userId,
134         String event,
135         @DelegatesTo(type = "com.agorapulse.micronaut.segment.builder.MessageBuilderWithProperties", strategy = Closure.DELEGATE_FIRST)
136         @ClosureParams(value = FromString.class, options = "com.agorapulse.micronaut.segment.builder.MessageBuilderWithProperties")
137             Closure<?> builder
138     ) {
139         self.track(userId, event, ConsumerWithDelegate.create(builder));
140     }
141 
142 }
